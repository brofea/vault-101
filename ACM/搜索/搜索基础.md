# 一、DFS

> [OI Wiki - DFS（搜索）](https://oi-wiki.org/search/dfs/)  

|     | 数据结构  | 空间需求    | 区别     | 适用于           | 重点    |
| --- | ----- | ------- | ------ | ------------- | ----- |
| DFS | stack | O (n)   | 不具有最短性 | 思路复杂的，最空间要求高的 | 回溯和减枝 |
| BFS | queue | O (2^n) | 最短路特性  | 最短最少最小        |       |

### DFS 例题 1 全排列问题

> [落谷 - P1706 全排列问题](https://www.luogu.com.cn/problem/P1706)  

```C++
\#include <iostream>
\#include <vector>
using namespace std;
int n,sta[15];
vector<bool> flag(15,0);
void dfs(int depth){
    if(depth>=n){
        for(int i=0;i<n;i++)
            printf("%5d",sta[i]);
        printf("\n");
        return;
    }
    for(int i=1;i<=n;i++){
        if(flag[i]==0){
            flag[i]=1;
            sta[depth]=i;
            dfs(depth+1);
            flag[i]=0;
        }
    }
    return;
}
int main(){
    cin>>n;
    dfs(0);
}
```

- DFS 如果要记录路径，可以将 `flag` 数组声明为`int`类型并初始化为`-1` ，在递归中记录`depth`

### DFS 例题 2 八皇后

> [落谷 - P1219 [USACO1.5] 八皇后 Checker Challenge](https://www.luogu.com.cn/problem/P1219)  

思路 1，定义`map[][]`数组，一个一个格子进行DFS，需要传递 `x`，`y` 作为遍历深度/位置。

思路 2，由于每行仅仅能存放一个皇后，可以将问题转化为数组全排列问题，比如`a[3]=6`代表存放在第三行第六列

两种思路都需要存放三种条件，分别是：列是否被使用`ifUsed`，左斜对角线是否被使用`ifUsedL`，右斜对角线是否被使用`ifUsedR`。对于思路2`ifUsedL`和`ifUsedR`的使用相当于剪枝，不需要枚举完所有的情况再判断是否有对角线冲突，而对于思路1，三个数组都是剪枝。

以下代码使用思路 2`initialize()`函数用于事先计算某行某列属于第几个左右对角线，减少DFS内部判断

```C++
\#include <iostream>
\#include <vector>
using namespace std;
const int MAX=13;
int n;
vector<bool> ifUsed(MAX+1,0);
vector<bool> ifUsedL(MAX*2,0);
vector<bool> ifUsedR(MAX*2,0);
int lline[MAX+1][MAX+1];
int rline[MAX+1][MAX+1];
void initialize(){
    for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            lline[i][j]=n+i-j;
            rline[i][j]=i+j-1;
        }
    }
}
int current[MAX+1];
int ptime=3;
int ans=0;
void dfs(int layer){
    if(layer>n){
        if(ptime>0){
            for(int i=1;i<=n;i++)
                cout<<current[i]<<' ';
            cout<<endl;
            ptime--;
        }
        ans++;
        return;
    }
    for(int i=1;i<=n;i++){
        if (ifUsed[i]==0 &&
            ifUsedL[lline[layer][i]]==0 &&
            ifUsedR[rline[layer][i]]==0){
            ifUsed[i]=ifUsedL[lline[layer][i]]=ifUsedR[rline[layer][i]]=1;
            current[layer]=i;
            dfs(layer+1);
            ifUsed[i]=ifUsedL[lline[layer][i]]=ifUsedR[rline[layer][i]]=0;
        }
    }
    return;
}
int main() {
    cin>>n;
    initialize();
    dfs(1);
    cout<<ans<<endl;
}
```

剪枝有两种，可行性剪枝与最优性剪枝，上述剪枝属于可行性剪枝

# 二、BFS

> [OI Wiki - BFS（搜索）](https://oi-wiki.org/search/bfs/)  

- Dp 是一种无环最短路问题，最短路问题包含了 Dp

BFS 运算时使用的数据结构是队列，在求最短路问题中时，先将起点入队，开始迭代时，先将队头出队，然后将符合结果的点入队，当对列为空时停止循环

BFS 与 DFS 都是一种思想，没有固定的框架，以下时 BFS 的经典例题

### BFS 例题 1 马的遍历

> [落谷 - P1443 马的遍历](https://www.luogu.com.cn/problem/P1443)  


```C++
\#include <iostream>
\#include <cstring>
\#define f first
\#define s second
using namespace std;
const int N=410;
int map[N][N];
int n,m,x,y;
// use [1][1] as point (1,1)
pair<int,int> q[N*N];
void bfs(){
    map[x][y]=0;
    q[0].f=x;
    q[0].s=y;
    int qhead=0,qtail=0;
    int xmove[8]={2,2,-2,-2,1,-1,1,-1};
    int ymove[8]={1,-1,1,-1,2,2,-2,-2};
    while(qhead<=qtail){
        pair<int,int> tmp=q[qhead++];
        for(int i=0;i<8;i++){
            int x=tmp.f+xmove[i];
            int y=tmp.s+ymove[i];
            if(x<=n&&x>=1&&y<=m&&y>=1&&map[x][y]==-1){
                map[x][y]=map[tmp.f][tmp.s]+1;
                q[++qtail].first=x;
                q[qtail].second=y;
            }
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++)
            printf("%-5d",map[i][j]);
        printf("\n");
    }
}
int main() {
    cin>>n>>m>>x>>y;
    memset(map,-1,sizeof map);
    bfs();
}
```

`qhead` 和 `qtail` 用于存放队头和队尾的位置

也可以用 STL queue 代替手搓的对列——

```C++
\#include <iostream>
\#include <cstring>
\#include <queue>
using namespace std;
const int N=410;
int map[N][N],n,m,x,y;
// use [1][1] as point (1,1)
queue<pair<int,int> > q;
void bfs(){
    map[x][y]=0;
    q.push(make_pair(x,y));
    int xmove[8]={2,2,-2,-2,1,-1,1,-1};
    int ymove[8]={1,-1,1,-1,2,2,-2,-2};
    while(!q.empty()){
        pair<int,int> tmp=q.front();q.pop();
        for(int i=0;i<8;i++){
            int x=tmp.first+xmove[i];
            int y=tmp.second+ymove[i];
            if(x<=n&&x>=1&&y<=m&&y>=1&&map[x][y]==-1){
                map[x][y]=map[tmp.first][tmp.second]+1;
                q.push(make_pair(x,y));
            }
        }
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++)
            printf("%-5d",map[i][j]);
        printf("\n");
    }
}
int main() {
    cin>>n>>m>>x>>y;
    memset(map,-1,sizeof map);
    bfs();
}
```

无论使用STL还是手动实现的对列，要记得获取队头数据后将其出队，在手动实现中即为`qhead++`

### BFS 例题 2 填涂颜色

> [落谷 - P1162 填涂颜色](https://www.luogu.com.cn/problem/P1162)  
>  

基本思路——

先将可填涂的格子赋值为-1，将不可填涂的格子赋值为-2，将 $i$ 行 `j` 列的点记作`id=i*n+j+1` ，遍历每个点，用其 id 开始填涂，和该点相接的色块全都为该点的 id。

在搜索完成之后`recolor_print()`函数遍历四周，找到和四周相邻的 id 的色块将其剔除出可用范围，最后检测可用且存在于`map`数组中的色块，将其赋值为2，将不可填涂的格子赋值为1，最后输出答案

```C++
\#include <iostream>
\#include <queue>
\#include <cstring>
using namespace std;
int n,map[35][35];
const int xmove[4]={1,-1,0,0};
const int ymove[4]={0,0,1,-1};
void bfs(int X,int Y,int id){
    queue<pair<int,int> > q;
    q.push(make_pair(X,Y));
    map[X][Y]=id;
    while(!q.empty()){
        pair<int,int> tmp=q.front();q.pop();
        for(int i=0;i<4;i++){
            int x=tmp.first+xmove[i];
            int y=tmp.second+ymove[i];
            if(x>=0&&x<n&&y>=0&&y<n&&map[x][y]==-1){
                map[x][y]=id;
                q.push(make_pair(x,y));
            }
        }
    }
}
void initailize(){
    cin>>n;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            scanf("%d",&map[i][j]);
    int wall=n*n+1;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            map[i][j]=map[i][j]==0?-1:wall;
    // use n*n+1 represent wall
}
void recolor_print(){
    // search the side of the map
    // see if there are some id torch the side
    bool flag[1000];
    // use flag store if the id can be turn into 2
    memset(flag,1,sizeof flag);
    for(int i=0;i<n;i++){
        flag[map[0][i]]=0;
        flag[map[n-1][i]]=0;
        flag[map[i][0]]=0;
        flag[map[i][n-1]]=0;
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(map[i][j]==n*n+1)
                map[i][j]=1;
            else if (flag[map[i][j]]==1)
                map[i][j]=2;
            else
                map[i][j]=0;
        }
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++)
            printf("%-2d",map[i][j]);
        printf("\n");
    }
}
int main() {
    initailize();
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            if(map[i][j]==-1)
                bfs(i,j,i*n+j+1);  
    recolor_print();
}
```

### BFS 例题 3 奇怪的电梯

> [落谷 - P1135 奇怪的电梯](https://www.luogu.com.cn/problem/P1135)  
>  

本题是一维 BFS 不需要用 pair 存储点的数据，较为典型的题目

```C++
\#include <iostream>
\#include <queue>
\#include <cstring>
using namespace std;
const int N=205;
int n,a,b,k[N],step[N];
queue<int> q;
int main() {
    memset(step,-1,sizeof step);
    cin>>n>>a>>b;
    for(int i=1;i<=n;i++)
        cin>>k[i];
    step[a]=0;
    q.push(a);
    while(!q.empty()){
        int tmp=q.front();q.pop();
        int depth=step[tmp];
        int up=tmp+k[tmp];
        int dn=tmp-k[tmp];
        if(up>=1&&up<=n&&step[up]==-1){
            q.push(up);
            step[up]=depth+1;
        }
        if(dn>=1&&dn<=n&&step[dn]==-1){
            q.push(dn);
            step[dn]=depth+1;
        }
    }
    cout<<step[b]<<endl;
}
```
